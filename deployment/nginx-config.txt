server {
    listen 80;
    server_name 15.206.74.151; # Updated IP

    client_max_body_size 20M; # Increased for multiple document uploads

    location / {
        root /home/ubuntu/Library_NITT/frontend/dist;
        try_files $uri /index.html;
        # Added cache busting for new deployments
        add_header Cache-Control "no-store, no-cache, must-revalidate";
    }

    location /api {
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        
        # These will handle the long PDF generation times
        proxy_connect_timeout 300;
        proxy_send_timeout 300;
        proxy_read_timeout 300;
        send_timeout 300;
    }
